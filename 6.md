## 1.Lru 算法
* lru：lru全称为最近最少使用算法，每次从cache中移除数据时间选择最长时间未使用。
* 算法实现：用链表实现极客，添加数据时，只需将最近使用的数据添加到链表的头部，如果删除数据时则将链表的尾部删除数据即可。


算法实现如下：
```
typedef struct DouListNode{
    struct DouListNode * prev;
    struct DouListNode * next;
    int val;
    ouListNode(int x){
        this->val = x;
        this->prev = NULL;
        this->next = NULL;
    }
}DouListNode;

class lru{
public:    
    lru(int x){
        capbility = x;
        this->head = NULL;
        this->tail = NULL;
    }

    bool insert(int val){
        if(cache.find(val) == cache.end()){
            DouListNode * node = new DouListNode(val);
            this->length++;

            if(!head){
                head = node;
                head->prev = NULL;
                tail = node;
                tail->next = NULL;
                return true;
            }

            if(this->length > capbility){
                DouListNode * t = this->tail;
                tail = tail->prev;
                tail->next = NULL;

                /*we move the last element*/
                cache.erase(t->val);
                free(t);
            }
            
            /*we add the first node at the first*/
            node->next = head;
            node->prev = NULL;
            if(head){
                head->prev = node;
            }
            head = node;
            cache[val] = node;

        }else{
            DouListNode * node = cache[val];
            if(head == node){
                return true;
            }
            if(tail == node){
                tail = node->prev;
            }
            if(node->prev){
                node->prev->next = node->next;
            }
            if(node->next){
                node->next->prev = node->prev;
            }

            /*we just move  node to the head */
            node->prev = NULL;
            node->next = head;
            if(head){
                head->prev = node;
            }
        }
    }

private:
    map<int,DouListNode *> cache; /*cache address*/
    int capbility;/*cache size*/
    int length;
    DouListNode * head;
    DouListNode * tail;
}
```

## 2.回文字符串
* 问题：Given a singly linked list, determine if it is a palindrome.

* 解法：将链表的前半段倒置即可，判断两个链表内容是否相同

```
typedef sturct ListNode{
    int val;
    sturct ListNode * prev;
    sturct ListNode * next;
}ListNode;

int lengthOfList(ListNode * head){
    int ans = 0;

    while(head){
        head = head->next;
        ans++;
    }

    return ans;
}

ListNode * reverse(ListNode * head){
    if(!head||!head->next){
        return head;
    }

    ListNode * prev = reverse(head->next);
    head->next->next = head;

    return prev;
}

bool isPalindrome(ListNode * head){
    if(!head||!head->next){
        return true;
    }

    ListNode * prev = NULL;
    ListNode * slow = NULL;
    ListNode * fast = NULL;
    int length = lengthOfList(head);

    while(fast&&fast->next){
        prev = slow;
        slow = slow->next;
        fast = fast->next;
        if(fast){
            fast = fast->next;
        }
    }

    prev->next = NULL;
    head = reverse(head);
    if(length%2){
        slow = slow->next;
    }

    while(slow&&head){
        if(head->val != slow->val){
            return false;
        }
        head = head->next;
        slow = slow->next;
    }
    
    return true;
}
```
