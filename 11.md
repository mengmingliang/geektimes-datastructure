bool selectSort(vector<int> &array){
    int n = array.size();

    for(int i = n-1;i >= 1; --i){
        for(int j = 0;j < i; ++j){
            if(array[j] > array[i]){

## 6.链表归并排序
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode * mergeList(ListNode * l1,ListNode * l2){
        if(!l1){
            return l2;
        }    
        
        if(!l2){
            return l1;
        }
        
        if(l1->val < l2->val){
            l1->next = mergeList(l1->next,l2);
            return l1;
        }else{
            l2->next = mergeList(l1,l2->next);
            return l2;
        }
    }
    
    ListNode* sortList(ListNode* head) {
        ListNode * slow = head;
        ListNode * fast = head;
        ListNode * tail = head;
        
        if(head == NULL || head->next == NULL){
            return head;
        }
        
        do{
            tail = slow;
            slow = slow->next;
            fast = fast->next;
            if(fast){
                fast = fast->next;
            }
        }while(fast&&slow);
        tail->next = NULL;
        
        ListNode * l1 = sortList(head);
        ListNode * l2 = sortList(slow);
        
        
        return mergeList(l1,l2);        
    }
};
