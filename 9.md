队列
====
## 1.循环队列实现
```
class queue{
public:
    queue(int x){
        this->capbility = x;
        this->item = new int[x];
        this->head = 0;
        this->tail = 0;
    }

    bool enqueue(int val){
        if(full()){
            return false;
        }

        tail = (tail+1)%capbility;
        item[tail] = val;

        return true;
    }

    bool pop(){
        if(empty()){
            return false;
        }

        head = (head+1)%capbility;
        return true;
    }


    int front(){
        return item[head];
    }

    int size(){
        return  (capbility - head + tail)%capbility;
    }

    void dump(){
        for(int i = head; i != tail; i = (i+1)%capbility){
            cout<<items[i]<<endl;
        }
    }

    bool empty(){
        if(head == tail){
            return true;
        }else{
            return false;
        }
    }

    bool full(){
        if((tail+1)%capbility == head){
            return true;
        }else{
            return false;
        }
    }

    bool free(){
        free(items);
    }

priavte:
    int capbility;
    int * items;
    int head;
    int tail;
}
```

## 2.无锁队列
    常见队列在生产和消费同时访问内存时，为了防止内存出现重入的问题，一般情况下需要加锁来防止内存重入，但是加锁会造成资源消耗过大，从而导致队列的访问速度下降，目前在许多需要高并发的情况时，一般采用无锁队列来实现。常见的有以下四种模式：

* 单生产者单消费者

    `典型的实现可以参考内核的kfifo的实现:`
```
typedef unsigned int uint32;
#define MAX_QUEUE_SIZE 0x80000000UL

class queue{
public:
    queue(uint32 x){
        if(x > 0x80000000UL){
            this->capbility = 0;
            this->head = 0;
            this->tail = 0;
            this->item = NULL;
            return;
        }

        if(!is_power_of_2(x)){
            x = roundup_pow_of_two(x);
        }

        this->capbility = x;
        this->item = new int[this->capbility];
        this->head = 0;
        this->tail = 0;
    }

    static bool is_power_of_2(uint32 x){
        return (x != 0) && ((x & (x - 1)) == 0);
    }

    static unsigned int roundup_pow_of_two(uint32 x){
        uint32 b = 0;

        for(int i = 0; i < 32; i++){
            b = 1UL << i;
            if(x <= b){
                 break;
            }
        }

        return b;
    }

    bool enqueue(int val){
        if(full()){
            return false;
        }

        int tail_next = (tail+1)%capbility;
        item[tail_next] = val;
        tail = tail_next;

        return true;
    }

    bool pop(){
        if(empty()){
            return false;
        }

        int head_next = (head+1)%capbility;
        head = head_next;

        return true;
    }


    int front(){
        return item[head];
    }

    int size(){
        return  (capbility - head + tail)%capbility;
    }

    void dump(){
        for(int i = head; i != tail; i = (i+1)%capbility){
            cout<<items[i]<<endl;
        }
    }

    bool empty(){
        if(head == tail){
            return true;
        }else{
            return false;
        }
    }

    bool full(){
        if((tail+1)%capbility == head){
            return true;
        }else{
            return false;
        } 
    }

    bool free(){
        free(items);
    }

priavte:
    uint32 capbility;
    uint32 * items;
    uint32 int;
    uint32 out;
    spinlock_t *lock;
}
```

```
#define MAX_QUEUE_SIZE 0x80000000UL
#define safe_cas(ptr,oldval,newval)  __atomic_compare_exchange((ptr),(oldval),(newval))

typdef unsigned char  uint8
typdef unsigned short uint16
typdef unsigned int   uint32
typdef unsigned long long uint64

class queue
{
public:
    queue(int x)
    {
        this->head = 0;
        this->tail = 0;
        this->capbility = 0;
        this->item = NULL;

        if(x > MAX_QUEUE_SIZE){
            return;
        }

        if(!is_power_of_2(x)){
            x = roundup_pow_of_two(x);
        }

        this->capbility = x;
        this->item = new int[this->capbility];
    }

    static bool is_power_of_2(uint32 x)
    {
        return (x != 0) && ((x & (x - 1)) == 0);
    }

    static uint32 roundup_pow_of_two(uint32 x)
    {
        uint32 b = 0;

        for(int i = 0; i < 32; i++){
            b = 1UL << i;
            if(x <= b){
                 break;
            }
        }

        return b;
    }

    /*the same funtction as roundup_pow_of_two*/
    static inline int fls(int x)
    {
        int r;
        __asm__("bsrl %1,%0\n\t"
                "jnz 1f\n\t"
                "movl $-1,%0\n"
                "1:" : "=r" (r) : "rm" (x));

        return r+1;
    }


    bool enQueue(uint32 val)
    {
        if(full()){
            return false;
        }

        uint32 tail_next = 0;

        do{
            tail_next = tail;
            /*we check if the queue is full*/
            if((tail + 1)%capbilty == head){
                return false;
            }
        }while(!safe_cas(&tail,tail_next,(tail_next+1)%capbility));

        tail_new = (tail_next+1)%capbility;
        this->item[tail_new] = val;
        safe_cas(tail,tail_next,tail_new);

        return true;
    }

    int outQueue()
    {
        if(empty()){
            return -1;
        }

        uint32 head_next = 0;

        do{
            head_next = head;
            /*we check if the queue is null*/
            if(head == tail){
                return -1;
            }
        }while(!safe_cas(&head,head_next,(head_next+1)%capbility));

        rt_val = this->item[head_next];

        return rt_val;
    }

    uint32 size()
    {
        uint32  tail_next = tail;
        uint32  tail_next = tail;

        return (tail - head + capbilty)%capbilty;
    }

    bool full()
    {
        return (tail + 1)%capbilty == head;
    }

    bool empty()
    {
        return this->head == this->tail;
    }

    bool dumpQueue()
    {
        for(int i = head; i != tail; i = (i+1)%capbility){
            cout<<item[i]<<endl;
        }
    }

    bool free()
    {
        free(items);
    }

private:
    uint32 * item;
    volatile uint32 head;
    volatile uint32 tail;
    uint32 capbility;
    spinlock_t *lock;
};
```

```
#define MAX_QUEUE_SIZE 0x80000000UL
#define safe_cas(ptr,oldval,newval)  __sync_bool_compare_and_swap((ptr),(oldval),(newval))
#define safe_add4(val, off)  __sync_fetch_and_add(val, off)
#define safe_sub4(val, off)  __sync_fetch_and_sub(val, off)

typdef unsigned char  uint8
typdef unsigned short uint16
typdef unsigned int   uint32
typdef unsigned long long uint64
typdef struct ListNode{
    int val;
    ListNode * next;
    ListNode(int x){
        this->val = x;
        this->next = NULL;
    }
}

class queue
{
public:
    queue()
    {
        this->head = NULL;
        this->tail = NULL;
        this->in = 0;
        this->out = 0;
    }

    bool enQueue(uint32 val)
    {
        ListNode * node = new ListNode(val);
      
        do{
            node->next = tail;
        }while(tail&&!safe_cas(&tail,node->next,node);
        safe_add4(&this->in);

        return true;
    }

    int outQueue()
    {
        do{
            ListNode * node = head;
            if(!node){
                return -1;
            }
        }while(head&&!safe_cas(&head,node,node->next);
        safe_add4(&this->out);
        free(node);

        return node->val;
    }

    uint32 size()
    {
        return this->in - this->out;
    }

    bool dumpQueue()
    {
        ListNode * head_curr = head;
        
        while(head_curr){
            if(head_curr){
                cout<<head_curr->val<<endl;
            }
        }
    }

private:
    volatile ListNode * head;
    volatile ListNode * tail;
    volatile uint in;
    volatile uint out;
    spinlock_t *lock;
};
```
