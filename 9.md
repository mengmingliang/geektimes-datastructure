## 1.循环队列实现
```
class queue{
public:
    queue(int x){
        this->capbility = x;
        this->item = new int[x];
        this->head = 0;
        this->tail = 0;
    }

    bool enqueue(int val){
        if(full()){
            return false;
        }

        tail = (tail+1)%capbility;
        item[tail] = val;

        return true;
    }

    bool pop(){
        if(empty()){
            return false;
        }

        head = (head+1)%capbility;
        return true;
    }


    int front(){
        return item[head];
    }

    int size(){
        return  (capbility - head + tail)%capbility;
    }

    void dump(){
        for(int i = head; i != tail; i = (i+1)%capbility){
            cout<<items[i]<<endl;
        }
    }

    bool empty(){
        if(head == tail){
            return true;
        }else{
            return false;
        }
    }

    bool full(){
        if((tail+1)%capbility == head){
            return true;
        }else{
            return false;
        }
    }

    bool free(){
        free(items);
    }

priavte:
    int capbility;
    int * items;
    int head;
    int tail;
}
```
